<script>

const SUPABASE_URL = 'https://bfzrfnhifgmcwyvdsect.supabase.co';

const SUPABASE_KEY = 'PASTE_YOUR_ANON_KEY_HERE';


let allParcels = [];


async function loadParcels() {

  try {

    const res = await fetch(

      `${SUPABASE_URL}/rest/v1/parcels?select=*&order=overall_score.desc`,

      { headers: { 'apikey': SUPABASE_KEY, 'Authorization': `Bearer ${SUPABASE_KEY}` } }

    );

    allParcels = await res.json();

    filterAndRender();

  } catch(e) {

    document.getElementById('grid').innerHTML = '<div class="empty">Error loading data. Check your Supabase URL and key.</div>';

  }

}


function scoreColor(s) {

  if (s >= 9) return { bg: '#E1F5EE', text: '#0F6E56' };

  if (s >= 8) return { bg: '#E6F1FB', text: '#185FA5' };

  if (s >= 7) return { bg: '#FAEEDA', text: '#854F0B' };

  return { bg: '#F1EFE8', text: '#5F5E5A' };

}


function useColor(use) {

  const map = {

    'Cold Storage': { bg: '#E1F5EE', text: '#0F6E56' },

    'Solar Farm': { bg: '#FAEEDA', text: '#854F0B' },

    'Data Centre': { bg: '#E6F1FB', text: '#185FA5' },

    'Logistics / Warehouse': { bg: '#F1EFE8', text: '#5F5E5A' },

    'Residential': { bg: '#FBEAF0', text: '#72243E' },

    'Agricultural': { bg: '#EAF3DE', text: '#3B6D11' },

    'Industrial': { bg: '#EEEDFE', text: '#3C3489' }

  };

  return map[use] || { bg: '#F1EFE8', text: '#444441' };

}


function filterAndRender() {

  const q = document.getElementById('search').value.toLowerCase();

  const district = document.getElementById('district').value;

  const usecase = document.getElementById('usecase').value;

  const minscore = parseFloat(document.getElementById('minscore').value);


  let filtered = allParcels.filter(p => {

    if (q && !p.name.toLowerCase().includes(q) && !p.district.toLowerCase().includes(q) && !(p.soil_type||'').toLowerCase().includes(q) && !(p.top_use||'').toLowerCase().includes(q) && !(p.investment_thesis||'').toLowerCase().includes(q)) return false;

    if (district && p.district !== district) return false;

    if (usecase && p.top_use !== usecase) return false;

    if (minscore > 0 && p.overall_score < minscore) return false;

    return true;

  });


  document.getElementById('count-stat').textContent = `${filtered.length} parcels`;

  const grid = document.getElementById('grid');


  if (!filtered.length) {

    grid.innerHTML = '<div class="empty">No parcels match your filters.</div>';

    return;

  }


  grid.innerHTML = filtered.map((p, i) => {

    const sc = scoreColor(p.overall_score);

    const uc = useColor(p.top_use);

    return `<div class="card" onclick="showModal(${allParcels.indexOf(p)})">

      <div class="card-top">

        <div>

          <div class="card-name">${p.name}</div>

          <div class="card-meta">${p.district} · ${p.mandal} · ${p.area_acres} acres</div>

        </div>

        <div class="score-circle" style="background:${sc.bg};color:${sc.text}">${p.overall_score}</div>

      </div>

      <div class="card-tags">

        <span class="tag" style="background:${uc.bg};color:${uc.text}">${p.top_use}</span>

        <span class="tag" style="background:#F1EFE8;color:#5F5E5A">${p.soil_type}</span>

        <span class="tag" style="background:#E6F1FB;color:#185FA5">₹${(p.guidance_value_per_sqyd/1000).toFixed(0)}K/sqyd</span>

      </div>

      <div class="card-desc">${p.investment_thesis}</div>

      <div class="scores-row">

        <div class="score-item"><div class="score-label">Cold Storage</div><div class="score-val" style="color:#0F6E56">${p.cold_storage_score}</div></div>

        <div class="score-item"><div class="score-label">Solar</div><div class="score-val" style="color:#854F0B">${p.solar_score}</div></div>

        <div class="score-item"><div class="score-label">Data Centre</div><div class="score-val" style="color:#185FA5">${p.data_centre_score}</div></div>

      </div>

    </div>`;

  }).join('');

}


function showModal(index) {

  const p = allParcels[index];

  const sc = scoreColor(p.overall_score);

  const uc = useColor(p.top_use);

  document.getElementById('modal-content').innerHTML = `

    <div class="modal-header">

      <div class="modal-title">${p.name}</div>

      <div class="modal-sub">${p.district} District · ${p.mandal} Mandal · ${p.area_acres} acres · ${p.soil_type}</div>

    </div>

    <div class="modal-body">

      <div class="modal-section">

        <div class="modal-section-title">Investment Thesis</div>

        <div class="thesis-box">

          <div class="thesis-label">TOP USE: ${p.top_use}</div>

          ${p.investment_thesis}

        </div>

      </div>

      <div class="modal-section">

        <div class="modal-section-title">Use Case Scores</div>

        <div class="scores-grid">

          ${[['Cold Storage','#0F6E56','#E1F5EE',p.cold_storage_score],['Solar Farm','#854F0B','#FAEEDA',p.solar_score],['Data Centre','#185FA5','#E6F1FB',p.data_centre_score],['Logistics','#5F5E5A','#F1EFE8',p.logistics_score],['Residential','#72243E','#FBEAF0',p.residential_score],['Agricultural','#3B6D11','#EAF3DE',p.agricultural_score]].map(([label,color,bg,val]) => `

          <div class="score-card" style="background:${bg};color:${color}">

            <div class="score-card-label">${label}</div>

            <div class="score-card-val">${val}</div>

            <div class="score-card-bar"><div class="score-card-fill" style="width:${val*10}%"></div></div>

          </div>`).join('')}

        </div>

      </div>

      <div class="modal-section">

        <div class="modal-section-title">Land Intelligence</div>

        <div class="info-grid">

          <div class="info-item"><div class="info-label">NH Distance</div><div class="info-val">${p.nh_distance_km} km</div></div>

          <div class="info-item"><div class="info-label">Power Substation</div><div class="info-val">${p.substation_33kv_km} km (33kV)</div></div>

          <div class="info-item"><div class="info-label">Flood Risk</div><div class="info-val">${['','Very Low','Low','Moderate','High','Very High'][p.flood_risk_score] || p.flood_risk_score}</div></div>

          <div class="info-item"><div class="info-label">Groundwater Depth</div><div class="info-val">${p.groundwater_depth_m}m</div></div>

          <div class="info-item"><div class="info-label">Annual AQI</div><div class="info-val">${p.aqi_annual} (${p.aqi_annual < 50 ? 'Good' : p.aqi_annual < 100 ? 'Moderate' : 'Poor'})</div></div>

          <div class="info-item"><div class="info-label">Railway Distance</div><div class="info-val">${p.railway_distance_km} km</div></div>

          <div class="info-item"><div class="info-label">Nearest School</div><div class="info-val">${p.school_nearest_km} km</div></div>

          <div class="info-item"><div class="info-label">Nearest Hospital</div><div class="info-val">${p.hospital_nearest_km} km</div></div>

          <div class="info-item"><div class="info-label">Guidance Value</div><div class="info-val">₹${p.guidance_value_per_sqyd.toLocaleString()}/sqyd</div></div>

          <div class="info-item"><div class="info-label">Price Trend (24m)</div><div class="info-val" style="color:${p.price_trend_24m_pct > 20 ? '#0F6E56' : '#444'}">+${p.price_trend_24m_pct}%</div></div>

          <div class="info-item"><div class="info-label">Current Use</div><div class="info-val">${p.current_use}</div></div>

          <div class="info-item"><div class="info-label">Overall Score</div><div class="info-val" style="color:${sc.text};font-size:18px;font-weight:700">${p.overall_score}</div></div>

        </div>

      </div>

      <div class="modal-section">

        <div class="modal-section-title">Data Confidence</div>

        <div class="confidence-bar">

          <div class="conf-track"><div class="conf-fill" style="width:${p.data_confidence}%"></div></div>

          <div class="conf-label">${p.data_confidence}%</div>

        </div>

      </div>

      <button class="modal-close" onclick="document.getElementById('modal').style.display='none'">Close</button>

    </div>`;

  document.getElementById('modal').style.display = 'block';

}


function closeModal(e) {

  if (e.target === document.getElementById('modal')) {

    document.getElementById('modal').style.display = 'none';

  }

}


loadParcels();
